(iii) HelloWorld Code

;Set 64-bit code
bits 64
global start
start:
jmp h
write:
;write(0x1, "Hello world!", 0xa)
;write() -> 4
mov r8b, 0x2
shl r8, 0x18
or r8, 0x4
mov rax, r8
xor rdi, rdi
mov dil, 0x1
pop rsi
xor rdx, rdx
mov dl, 0xe
syscall
sub r8b, 0x3
mov rax, r8
xor rdi, rdi
syscall
h:
call write
db "Hello, world!", 10
